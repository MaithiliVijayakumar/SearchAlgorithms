#include <stdio.h>

int british_museum_search(const int *arr, int n, int target) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            return i; // Found
        }
    }
    return -1; 
}

int main() {
    // Properly declared array
    int arr[] = {3, 7, 2, 9, 5, 8, 1, 6, 4, 0};
    int n = sizeof(arr) / sizeof(arr[0]);
    int target;

    printf("Enter the number to search: ");
    if (scanf("%d", &target) != 1) {
        printf("Invalid input.\n");
        return 1;
    }

    int index = british_museum_search(arr, n, target);

    if (index != -1)
        printf("Element %d found at index %d\n", target, index);
    else
        printf("Element %d not found\n", target);

    return 0;
}
