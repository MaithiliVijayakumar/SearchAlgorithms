#include <stdio.h>

#define MAX 100

int visited[MAX];
int history[MAX];
int historyCount = 0;

void DFS(int graph[MAX][MAX], int n, int node) {
    visited[node] = 1;
    history[historyCount++] = node; 

    for (int i = 0; i < n; i++) {
        if (graph[node][i] && !visited[i]) {
            DFS(graph, n, i);
        }
    }
}

int main() {
    int n = 7; 
    int graph[MAX][MAX] = {0};

    graph[0][1] = graph[1][0] = 1;
    graph[0][2] = graph[2][0] = 1;
    graph[1][3] = graph[3][1] = 1;
    graph[1][4] = graph[4][1] = 1;
    graph[2][5] = graph[5][2] = 1;
    graph[2][6] = graph[6][2] = 1;
    for (int i = 0; i < n; i++) {
        visited[i] = 0;
    }

    int startNode = 0;
    DFS(graph, n, startNode);
    printf("DFS Expansion Order: ");
    for (int i = 0; i < historyCount; i++) {
        printf("%d", history[i]);
        if (i < historyCount - 1) printf(" -> ");
    }
    printf("\n");

    return 0;
}
